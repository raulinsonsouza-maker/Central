    <script>
        function exportarParaPNG() {
            const painel = document.getElementById('dashboard-cmo');
            html2canvas(painel, {
                backgroundColor: '#000000',
                scale: 2,
                windowWidth: 1200 // Força uma largura boa para apresentação
            }).then(canvas => {
                const link = document.createElement('a');
                link.download = 'Report-Executivo-Ads.png';
                link.href = canvas.toDataURL('image/png');
                link.click();
            });
        }

        document.addEventListener('DOMContentLoaded', function() {
            const ctx = document.getElementById('cmoChart').getContext('2d');

            const data = {
                labels: ['Distribuidora (Ampla)', 'Insumos (Ampla)'],
                datasets: [
                    {
                        label: 'Volume de Leads',
                        data: [8, 4],
                        backgroundColor: '#e60000', // Vermelho V4 para volume
                        borderRadius: 4,
                        barPercentage: 0.6,
                        yAxisID: 'y'
                    },
                    {
                        label: 'CPA (R$)',
                        data: [29.88, 12.45],
                        backgroundColor: '#15803d', // Verde V4 para custo (eficiência)
                        borderRadius: 4,
                        barPercentage: 0.6,
                        yAxisID: 'y'
                    }
                ]
            };

            const config = {
                type: 'bar',
                data: data,
                options: {
                    indexAxis: 'y', // Gráfico horizontal (melhor para ler nomes de campanhas)
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: {
                            position: 'bottom',
                            labels: { color: '#cbd5e1', font: { family: "'Inter', sans-serif", size: 12 }, usePointStyle: true }
                        },
                        tooltip: {
                            backgroundColor: '#161616',
                            titleColor: '#fff',
                            bodyColor: '#ccc',
                            borderColor: '#333',
                            borderWidth: 1,
                            padding: 10,
                            callbacks: {
                                label: function(context) {
                                    if (context.dataset.label === 'CPA (R$)') {
                                        return 'CPA: R$ ' + context.parsed.x.toFixed(2);
                                    }
                                    return context.dataset.label + ': ' + context.parsed.x;
                                }
                            }
                        }
                    },
                    scales: {
                        x: {
                            grid: { color: '#333', drawBorder: false },
                            ticks: { color: '#94a3b8' }
                        },
                        y: {
                            grid: { display: false, drawBorder: false },
                            ticks: { color: '#f8fafc', font: { family: "'Inter', sans-serif", weight: '500' } }
                        }
                    }
                }
            };

            new Chart(ctx, config);
        });
    </script>
